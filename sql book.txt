CREATE TABLE TB_BOOK(
    BK_TITLE VARCHAR2(20) CONSTRAINT 이름오류 NOT NULL,
    BK_AUTHOR VARCHAR2(20)CONSTRAINT 저자오류 NOT NULL,
    BK_CODE VARCHAR2(20) CONSTRAINT 코드번호오류 PRIMARY KEY,
    BK_ISRENT NUMBER(1) DEFAULT 1 CONSTRAINT 렌트오류   NOT NULL 
);
--NUMBER(1)은 0 또는 1 값을 가질 수 있으며, 0은 false, 1은 true를 나타냅니다.
DROP TABLE TB_HUMAN ;

CREATE TABLE TB_HUMAN(
    HM_KEY NUMBER CONSTRAINT 키오류 PRIMARY KEY,
    HM_NAME VARCHAR2(20) CONSTRAINT 사람이름오류 NOT NULL,
    HM_RNUMBER VARCHAR2(20 ) CONSTRAINT 주민번호오류 NOT NULL,
    HM_AGE NUMBER ,
    HM_GENDER CHAR(6),
    HM_RENBOOKCODE VARCHAR2(20) DEFAULT 0 REFERENCES TB_BOOK NOT NULL   
);
